-- Create tables
CREATE TABLE Students
(
    neptunID CHAR(6) NOT NULL
        CHECK (neptunID LIKE 'OE%'),
    firstName VARCHAR(240) NOT NULL,
    lastName VARCHAR(240) NOT NULL,
    dateOfBirth DATE NOT NULL,
    email VARCHAR(240),
    CONSTRAINT students_pk PRIMARY KEY(neptunID)
);

CREATE TABLE Courses
(
    courseID NUMERIC(11) NOT NULL,
    name VARCHAR(240) NOT NULL,
    dateOfAnnounced DATE NOT NULL,
    credit NUMERIC(2)
        CHECK (credit > 0),
    classRoomID NUMERIC(11),
    requirementID NUMERIC(11),
    CONSTRAINT courses_pk PRIMARY KEY(courseID)
);

CREATE TABLE Teachers
(
    teacherID NUMERIC(11) NOT NULL,
    lastName VARCHAR(240) NOT NULL,
    firstName VARCHAR(240) NOT NULL,
    salary NUMERIC(6)
        CHECK (salary > 0),
    CONSTRAINT teacher_pk PRIMARY KEY(teacherID)
);

CREATE TABLE ClassRooms
(
    classRoomID NUMERIC(11) NOT NULL,
    capacity NUMERIC(11)
        CHECK (capacity > 0),
    CONSTRAINT classRooms_pk PRIMARY KEY(classRoomID)
);

CREATE TABLE GradeBook
(
    gradeBookID NUMERIC(11) NOT NULL,
    neptunID CHAR(6) NOT NULL,
    teacherID NUMERIC(11) NOT NULL,
    courseID NUMERIC(11) NOT NULL,
    grade NUMERIC(1)
        CHECK (grade BETWEEN 1 AND 5),
    rating NUMERIC(1)
        CHECK (rating BETWEEN 1 AND 5),
    CONSTRAINT leckekonyv_pk PRIMARY KEY(gradeBookID)
);
